```csharp
public void Collapse(string reason = "Unknown")
{
    AppLogger.Debug($"[FloatingIsland] Collapse() called - reason: {reason}, _isExpanded={_isExpanded}");
    if (!_isExpanded) 
    {
        AppLogger.Debug($"[FloatingIsland] Collapse() skipped ({reason}) - already collapsed");
        return;
    }
    _isExpanded = false;

    // UI 动作：切换 Visibility，保证动画开始前图标先出来
    IconGrid.Visibility = Visibility.Visible;
    FloatingPanel.Visibility = Visibility.Collapsed;

    // 停止正在进行的任何位移动画
    BeginAnimation(WidthProperty, null);
    BeginAnimation(HeightProperty, null);
    BeginAnimation(LeftProperty, null);
    BeginAnimation(TopProperty, null);

    // 计算收缩目标位置（图标原位置）
    double newLeft = _savedIconLeft;
    double newTop = _savedIconTop;

    // 设置过渡锁
    _isTransitioning = true;
    
    var widthAnim = new DoubleAnimation(50, new Duration(TimeSpan.FromMilliseconds(200)))
    { EasingFunction = new QuadraticEase { EasingMode = EasingMode.EaseOut } };
    
    // 动画完成回调：解除过渡锁并清理 UI 状态
    widthAnim.Completed += (s, e) => 
    { 
        _isTransitioning = false; 
        FloatingPanel.Visibility = Visibility.Collapsed; // 确保面板物理状态为 Collapsed
        AppLogger.Debug("[FloatingIsland] Collapse animation completed");
    };

    BeginAnimation(WidthProperty, widthAnim);
    BeginAnimation(HeightProperty, new DoubleAnimation(50, ...)); // 略去重复内容
    BeginAnimation(LeftProperty, new DoubleAnimation(newLeft, ...));
    BeginAnimation(TopProperty, new DoubleAnimation(newTop, ...));

    RestartInactiveTimer();
}
```

### 4. EnsureVisibleAndActivated()
UI 状态的防御性逻辑，核心修复即在此。
```csharp
// 兜底：确保窗口可见并激活，修正 UI 状态不一致
private void EnsureVisibleAndActivated()
{
    AppLogger.Debug($"[FloatingIsland] EnsureVisibleAndActivated() - Visibility={Visibility}");
    try
    {
        // 1. 物理显隐保护
        if (Visibility != Visibility.Visible)
        {
            Show();
        }

        // 2. 层级激活保护：置顶切换技巧强制系统重刷层级
        Activate();
        Topmost = true;
        Topmost = false;
        
        // 3. 核心修复：防止逻辑冲突导致的图标、面板同时存在或同时隐藏
        AppLogger.Debug("[FloatingIsland] EnsureVisible - Verifying Panel/Grid visibility");
        
        // 强制修正面板显示：展开过程必须保证面板是可见的
        if (FloatingPanel.Visibility != Visibility.Visible)
        {
            AppLogger.Debug("[FloatingIsland] EnsureVisible - Forcing FloatingPanel Visible");
            FloatingPanel.Visibility = Visibility.Visible;
        }
        
        // 强制修正图标隐藏：展开态不应显示小球
        if (IconGrid.Visibility != Visibility.Collapsed)
        {
            AppLogger.Debug("[FloatingIsland] EnsureVisible - Forcing IconGrid Collapsed");
            IconGrid.Visibility = Visibility.Collapsed;
        }

        UpdateLayout(); // 强制刷新布局渲染
    }
    catch (Exception ex)
    {
        AppLogger.Error("[FloatingIsland] EnsureVisibleAndActivated error", ex);
    }
}
```

```csharp
    BeginAnimation(TopProperty, null);
```

```csharp
private Point CalculateExpandPosition()
{
    Rect workArea = SystemParameters.WorkArea;

    double targetX = workArea.Right - MAX_WIDTH - 20;
    double targetY = workArea.Top + 60;

    if (_dockToMainWindow && MainWindowReference != null)
    {
        var mainLeft = MainWindowReference.Left;
        var mainTop = MainWindowReference.Top;
        var mainWidth = MainWindowReference.ActualWidth;
        var mainHeight = MainWindowReference.ActualHeight;

        bool isMaximized = MainWindowReference.WindowState == WindowState.Maximized;

        if (isMaximized)
        {
            // 全屏 → 向内部展开，贴靠右上角
            targetX = mainLeft + mainWidth - MAX_WIDTH - 20;
            targetY = mainTop + 20;
        }
        else
        {
            // 普通窗口 → 向外展开，浮窗贴靠窗口右上角
            targetX = mainLeft + mainWidth + 10;  // 向外偏移10px
            targetY = mainTop + 20;
        }

        // 确保不超出屏幕
        targetX = Math.Min(targetX, workArea.Right - MAX_WIDTH);
        targetX = Math.Max(targetX, workArea.Left);
        targetY = Math.Min(targetY, workArea.Bottom - MAX_HEIGHT);
        targetY = Math.Max(targetY, workArea.Top);
    }

    return new Point(targetX, targetY);
}
```

```csharp
}
```

# 二、拖拽 + 磁吸边缘

```csharp
private bool _isDragging = false;
private Point _dragOffset;

private void Window_MouseLeftButtonDown(object sender, MouseButtonEventArgs e)
{
    _isDragging = true;
    _dragOffset = e.GetPosition(this);
    CaptureMouse();
}

private void Window_MouseMove(object sender, MouseEventArgs e)
{
    if (!_isDragging) return;

    var pos = e.GetPosition(null);
    Left = pos.X - _dragOffset.X;
    Top = pos.Y - _dragOffset.Y;
}

private void Window_MouseLeftButtonUp(object sender, MouseButtonEventArgs e)
{
    _isDragging = false;
    ReleaseMouseCapture();
    SnapToScreenEdge();
}

private void SnapToScreenEdge()
{
    var screen = SystemParameters.WorkArea;

    // 左右贴边
    if (Math.Abs(Left - screen.Left) < 30) Left = screen.Left;
    if (Math.Abs(Left + Width - screen.Right) < 30) Left = screen.Right - Width;

    // 上下贴边
    if (Math.Abs(Top - screen.Top) < 30) Top = screen.Top;
    if (Math.Abs(Top + Height - screen.Bottom) < 30) Top = screen.Bottom - Height;
}
```
