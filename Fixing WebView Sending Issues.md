放心，**绝对不会丢失！**

因为刚才我在写架构的时候，特意把所有的数据存储位置都和程序本身（`exe` 或 `bin` 文件夹）**完全剥离开**了。

具体来说：
1. **你的所有配置（窗口排版位置、自定义网页地址）**：都存放在了系统级的隐藏缓存目录 `C:\Users\你的用户名\AppData\Local\WebViewHub\layout.json` 文件里。 
2. **你的网页缓存（Cookie、记住的密码、登录状态）**：也是存放在 `C:\Users\你的用户名\AppData\Local\WebViewHub\Profiles\` 下面。

这些数据都是写在你的 Windows 系统个人中心里的系统硬盘内部。这意味着：
- 不管你怎么**重新编译改代码**
- 不管你把软件产生的 `exe` **放到桌面上、D盘里，到处移动**
- 不管你是打开了 `Debug` 里的，还是打开了刚发的 `Release` 独立包

**只要是这台电脑，只要 Profile ID 还是同样的（Profile_1, Profile_2...），打开以后它们都会去系统同一个地方读取你的缓存，网页和排版都会原封不动地加载回来！** 所以你可以放心大胆地随便编译和测试。

核心原因就四个字：**万法归一 (Polyfill)**。

如果我们要做的产品是一个《Auto-ChatGPT-Client》，那么我100%会选择写个 Websocket hook 抓它的通讯流。可是我们现在在做的是 **WebViewHub（多模态聚合舱）**——你要把 Grok、Kimi、豆包、通义千问、Gemini 这些牛鬼蛇神全部在一个软件里跑通对话！

1. **协议孤岛灾难**：如果你顺着 **路线1** 走网络层拦截，你会发现 ChatGPT 是 SSE 事件流；豆包可能是私有的 Protobuf+WSS 双向流；Grok 可能是轮询拼合；Kimi 又是另一种 GraphQL 切片……你要为每一个大模型去逆向它们极其反人类的前端混淆协议，而且他们后端的接口一周一换，你每天都在修各种反爬虫字段，维护成本绝对灾难！
2. **DOM 是最终的收敛点**：我之所以死磕底层，选择“等待它们思考完毕后，直接暴力抓取最后一个带特定 class 的气泡元素的文本”，是因为无论后端的协议是加密的、切片的、打散的如何花里胡哨——**前端永远要把最终的字渲染到屏幕上的 HTML 容器里让用户看待。** DOM 树就是所有大厂工程师能作妖的终极尽头。
3. **性价比极致**：靠 `querySelector` 拿最后一段文本的容错率极高，哪怕网页大改版，改一改 CSS 选择器名字就行了，永远不用懂他们错综复杂的网络层通信机制。

# 多模型 Markdown 抽取器（增强版）

```javascript
(async function () {

  function detectPlatform() {

    const host = location.hostname;

    if (host.includes("chatgpt")) return "chatgpt";
    if (host.includes("gemini")) return "gemini";
    if (host.includes("kimi")) return "kimi";
    if (host.includes("grok")) return "grok";
    if (host.includes("doubao")) return "doubao";

    return "generic";
  }


  const adapters = {

    chatgpt() {
      return document.querySelectorAll(
        '[data-message-author-role="assistant"]'
      );
    },

    gemini() {
      return document.querySelectorAll(
        'model-response, .response-content'
      );
    },

    kimi() {
      return document.querySelectorAll(
        '.markdown-body'
      );
    },

    grok() {
      return document.querySelectorAll(
        '.message.assistant'
      );
    },

    doubao() {
      return document.querySelectorAll(
        '.ai-message'
      );
    },

    generic() {

      return document.querySelectorAll(`
        article,
        .assistant,
        .model-response,
        .markdown,
        .response
      `);

    }

  };


  function getLastAssistantNode() {

    const platform = detectPlatform();

    const finder = adapters[platform] || adapters.generic;

    const nodes = finder();

    if (!nodes || !nodes.length) return null;

    return nodes[nodes.length - 1];

  }


  function waitForStreamingEnd(node, idle = 700) {

    return new Promise(resolve => {

      let last = node.innerText;
      let stableTime = 0;

      const timer = setInterval(() => {

        const now = node.innerText;

        if (now === last) {
          stableTime += 100;
        } else {
          stableTime = 0;
        }

        if (stableTime >= idle) {
          clearInterval(timer);
          resolve();
        }

        last = now;

      }, 100);

    });

  }


  function forceRender(node) {

    node.scrollIntoView({
      block: "end",
      behavior: "instant"
    });

  }


  function cloneAndClean(node) {

    const clone = node.cloneNode(true);

    const trash = [
      "button",
      "svg",
      "img",
      ".copy",
      ".toolbar",
      ".actions",
      ".feedback",
      "[role='button']"
    ];

    trash.forEach(sel => {
      clone.querySelectorAll(sel).forEach(n => n.remove());
    });

    return clone;

  }


  function repairCodeBlocks(root) {

    root.querySelectorAll("pre").forEach(pre => {

      const code = pre.querySelector("code");

      if (!code) return;

      let lang = "";

      if (code.className) {

        const m = code.className.match(/language-(\w+)/);

        if (m) lang = m[1];

      }

      pre.setAttribute("data-lang", lang);

    });

  }


  function repairLatex(root) {

    root.querySelectorAll(".katex, .math").forEach(node => {

      const tex = node.innerText;

      const text = document.createTextNode(`$${tex}$`);

      node.replaceWith(text);

    });

  }


  function repairTables(root) {

    root.querySelectorAll("table").forEach(table => {

      if (!table.querySelector("thead")) {

        const first = table.querySelector("tr");

        if (!first) return;

        const thead = document.createElement("thead");

        thead.appendChild(first.cloneNode(true));

        table.insertBefore(thead, table.firstChild);

      }

    });

  }


  function buildTurndown() {

    const td = new TurndownService({
      codeBlockStyle: "fenced",
      headingStyle: "atx",
      hr: "---"
    });


    td.addRule("fencedCode", {

      filter: node => node.nodeName === "PRE",

      replacement: (content, node) => {

        const lang = node.getAttribute("data-lang") || "";

        const code = node.innerText;

        return `\n\`\`\`${lang}\n${code}\n\`\`\`\n`;

      }

    });


    td.addRule("tables", {

      filter: "table",

      replacement: function (content, node) {

        let md = "\n";

        const rows = node.querySelectorAll("tr");

        rows.forEach((tr, i) => {

          const cells = tr.querySelectorAll("th,td");

          const row = [...cells].map(c => c.innerText.trim());

          md += "| " + row.join(" | ") + " |\n";

          if (i === 0) {

            md += "| " + row.map(() => "---").join(" | ") + " |\n";

          }

        });

        return md + "\n";

      }

    });

    return td;

  }


  function htmlToMarkdown(html) {

    const turndown = buildTurndown();

    return turndown.turndown(html);

  }


  async function extract() {

    const node = getLastAssistantNode();

    if (!node) return "";

    forceRender(node);

    await waitForStreamingEnd(node);

    const clean = cloneAndClean(node);

    repairCodeBlocks(clean);
    repairLatex(clean);
    repairTables(clean);

    const html = clean.innerHTML;

    return htmlToMarkdown(html);

  }


  return await extract();

})();
```

# WebViewHub 最终架构会变成

```
WebViewHub
    │
ExecuteScriptAsync
    │
DOM Extractor
    │
HTML
    │
Turndown
    │
Markdown
    │
C#
```

这个架构的优点：

* **完全跨 AI**
* **不依赖网络协议**
* **不依赖剪贴板**
* **抗前端变化**
